JS基础（三） 条件循环语句

2019-09-15

1、条件语句

条件语句实际上是一种运算，这个运算最终判断是布尔值

        if(x || x===0){
            x++;
        }
        console.log(x);

        var  y=10;
//        y>0是true  y<20 是true   true && true  =true
        if(y>0 && y<20){
            y++;
        }
        console.log(y);


        var str="";
//        str 是""，它转换为布尔是false,!false=true
//        所以下面的条件是成立的
        if(!str){
            str="a";
        }
        console.log(str);

2、多支条件结构

switch (表达式){
            case 值1:
               ...
                break;
            case 值2:
               ...
                break;
            case 值3:
               ...
                break;
            case 值4:
               ...
                break;
            default:
               ...
                break;
        }
//switch(表达式),表达式是一个变量或者或者一个运算式,
//这个运算的结果或这个变量的值等于下面case中那一个,就会执行那个内容

3、while循环

跳出循环有两种形式：
1、循环条件不满足
2、当循环条件满足，通过break跳出循环

var i=0;
while(i<50){
			i++;
}

//死循环
/*
while(true){
		if(){
				break;
		}
}
*/
var s=0;
while(true){
		s++;
		if(s>50){
				break;
		}
}
//循环不能超出千万级以上，否则会造成卡顿状态，甚至卡死

//九九乘法表
  var i=1;
        var str="";
        while (i<10){
            var j=i;
            while (j<10){
                str+=i+"*"+j+"="+i*j+"&nbsp;";
//                document.write(i+"*"+j+"="+i*j+"&nbsp;");
                j++;
            }
//            document.write("|<br>");
            str+="<br>";
            i++;
     }
     document.write(str);

do{
